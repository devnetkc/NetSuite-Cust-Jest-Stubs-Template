/**
 * @alias module:rValidateSalesDesc
 * @function
 * @description Series of methods useful for running validations
 * @version 1.1.0
 * @param {Object} log
 * @param {Object} dialog
 */
function rValidateSalesDesc(log, dialog) {
  var salesDescId = 'salesdescription';
  /**
   *
   * @name saveRecord
   * @description Validates description length meets requirements before saving
   * @function
   * @param {Object} context - Script Context
   * @return {boolean}
   */
  function saveRecord(context) {
    // If the description has not been validated as good then do not save!
    return descVal_validDescription(context);
  }
  /**
   *
   * @name fieldChanged
   * @description Validates description length meets requirements when field is updated
   * @function
   * @param {Object} context - Script Context
   * @return {boolean}
   */
  function fieldChanged(context) {
    if (context.fieldId !== salesDescId) return;
    return descVal_validDescription(context);
  }
  /**
   *
   * @name descVal_validDescription
   * @function
   * @description Validates length of description meets requirements.
   * @param {Object} context - Script Context
   * @return {boolean}
   */
  function descVal_validDescription(context) {
    var validDescriptionLength = true;
    const Description = context.currentRecord.getText({
      fieldId: salesDescId,
    });
    const DescriptionLength = Description.length;
    if (DescriptionLength > 45) {
      // display dialog of error before stopping save function
      dialogOptions = {
        title: 'Description too long!',
        message:
          'The current description is too long in length.  To keep items to single lines we ask you keep the length to less than 45 characters.  Thanks!',
      };
      descVal_displayError(dialogOptions);

      validDescriptionLength = false;
    }
    log.debug({
      title: 'validDescription',
      details: { validDescriptionLength, Description, DescriptionLength },
    });
    return validDescriptionLength;
  }

  /**
   *
   * @name descVal_displayError
   * @function
   * @description Validates length of description meets requirements.
   * @param {Error|String} errMsg - Error object or string with error details included
   * @return {void}
   */
  function descVal_displayError(errMsg) {
    try {
      dialog.alert(errMsg);
      log.debug(errMsg);
    } catch (e) {
      log.error({
        title: e.name,
        details: e.message,
      });
    }
  }
  if (typeof module !== 'undefined' && typeof require !== 'undefined') {
    return {
      saveRecord,
      fieldChanged,
      descVal_validDescription,
      descVal_displayError,
    };
  }
  return {
    saveRecord,
    fieldChanged,
  };
}
